model
{
    for (i in 1:N) {
        y[i] ~ dlnorm(mu[i], sigma)
        mu[i] <- beta0 + beta1 * x1[i] + beta2 * x2[i] + b0[id[i]]
    }
    sigma ~ dunif(0.1, 100)
    beta0 ~ dunif(-5, 5)
    beta1 ~ dunif(-5, 5)
    beta2 ~ dunif(-5, 5)
    for (i in 1:Nsubj) {
        b0[i] ~ dnorm(0.00000E+00, 0.1)
    }
}
