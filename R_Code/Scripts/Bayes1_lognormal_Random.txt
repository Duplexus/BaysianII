model
{
    for (i in 1:N) {
        y[i] ~ dlnorm(mu[i], sigma)
        mu[i] <- beta1 * x1[i] + b0[id[i]]
        predict[i] ~ dlnorm(mu[i], sigma)
    }
    sigma ~ dunif(0.1, 100)
    beta1 ~ dnorm(0.00000E+00, 1.00000E-06)
    for (i in 1:Nsubj) {
        b0[i] ~ dnorm(0.00000E+00, 1.00000E-04)
    }
    sds <- pow(sd(b0[]), -1)
    for (i in 1:Nsubj) {
        mu_gr[i] <- mean(mu[i:(i + 6)])
        b0.rep[i] ~ dnorm(mu_gr[i], sds)
    }
}
